/opt/keycloak/bin/kcadm.sh config credentials --server http://localhost:8080 --realm master --user admin --password 123456
/opt/keycloak/bin/kcadm.sh create realms -s realm="%%KEYCLOAK_REALM%%" -s enabled=true
/opt/keycloak/bin/kcadm.sh create clients -r %%KEYCLOAK_REALM%% -s clientId=%%CLIENT_ID%% -s enabled=true -s publicClient=true -s directAccessGrantsEnabled=true -i
/opt/keycloak/bin/kcadm.sh create clients/%%CLIENT_UUID%%/roles -r %%KEYCLOAK_REALM%% -s name=%%ROLE_ADMIN%%
/opt/keycloak/bin/kcadm.sh create clients/%%CLIENT_UUID%%/roles -r %%KEYCLOAK_REALM%% -s name=%%ROLE_POWER_USER%%
/opt/keycloak/bin/kcadm.sh create users -r %%KEYCLOAK_REALM%% -s username=%%ADMIN_USERNAME%% -s enabled=true -s email=administrator@email.com -s firstName=Administrator -s lastName=User
/opt/keycloak/bin/kcadm.sh set-password -r %%KEYCLOAK_REALM%% --username %%ADMIN_USERNAME%% --new-password 123456
/opt/keycloak/bin/kcadm.sh add-roles -r %%KEYCLOAK_REALM%% --uusername %%ADMIN_USERNAME%% --cclientid %%CLIENT_ID%% --rolename %%ROLE_ADMIN%%
/opt/keycloak/bin/kcadm.sh create users -r %%KEYCLOAK_REALM%% -s username=%%POWER_USER_USERNAME%% -s enabled=true -s email=power_user@email.com -s firstName=Power -s lastName=User
/opt/keycloak/bin/kcadm.sh set-password -r %%KEYCLOAK_REALM%% --username %%POWER_USER_USERNAME%% --new-password 123456
